#' Calculer la distribution des √¢ges
#'
#' @param df Un `data.frame` contenant une colonne `Date.de.naissance` au format Date.
#' @return Un vecteur de quartiles (`0%, 25%, 50%, 75%, 100%`) des √¢ges.
#'
#' @export
calcul_distribution_age <- function(df) {
  df$Age <- as.numeric(difftime(Sys.Date(), df$Date.de.naissance, units = "days")) / 365.25
  quantile(df$Age, probs = c(0, 0.25, 0.5, 0.75, 1), na.rm = TRUE)
}

#' Compter le nombre d'adjoints
#'
#' @param df Un `data.frame` contenant la colonne `Libell√©.de.la.fonction`.
#' @return Un entier repr√©sentant le nombre d'adjoints trouv√©s.
#'
#' @export
compter_nombre_d_adjoints <- function(df) {
  sum(grepl("adjoint", df$Libell√©.de.la.fonction, ignore.case=TRUE), na.rm = TRUE)
}

#' Trouver l'√©lu(e) le plus √¢g√©(e)
#'
#' @param df Un `data.frame` contenant la colonne `Date.de.naissance`.
#' @return Un `data.frame` contenant l'√©lu(e) le plus √¢g√©(e).
#'
#' @export
trouver_l_elu_le_plus_age <- function(df) {
  df$Age <- as.numeric(format(Sys.Date(), "%Y")) - as.numeric(substr(df$Date.de.naissance, 1, 4))
  oldest <- df[which.max(df$Age), ]
  oldest[c("Nom.de.l.√©lu", "Pr√©nom.de.l.√©lu")]
}

#' R√©sum√© des statistiques d'une commune

#' @param x Un `data.frame` contenant `Libell√©.de.la.commune`, `Date.de.naissance`, etc.
#' @export

  cat("Commune:", unique(x$Libell√©.de.la.commune)[1], "\n")
  cat("Nombre d'√©lus:", compter_nombre_d_elus(x), "\n")
  cat("Distribution des √¢ges:\n")
  print(calcul_distribution_age(x))
  cat("√âlu(e) le plus √¢g√©(e):\n")
  print(trouver_l_elu_le_plus_age(x))
